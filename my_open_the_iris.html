def my_print_and_test_models(dataset):
  X=dataset.iloc[:,:-1]
  Y= dataset['class']
  X_train, X_test,y_train, y_test= train_test_split(X, Y, test_size=0.2)

  Tree_clf=tree.DecisionTreeClassifier()

  GNB_clf=GaussianNB()

  KN_clf =KNeighborsClassifier()

  LR_clf =LogisticRegression(solver='liblinear', multi_class='ovr')

  LD_clf =LinearDiscriminantAnalysis()

  SVM_clf = SVC(gamma='auto')


  models={"DecisionTree":Tree_clf, "GaussianNB":GNB_clf, "KNeighbors": KN_clf, "LogisticRegression":LR_clf, "LinearDiscriminant":LD_clf, "SVM":SVM_clf}

  for model_name, model_clf in models.items():
    cv_results = cross_val_score(model_clf, X_train, y_train, cv=None, scoring='accuracy')
    print(f"{model_name}: {round(cv_results.mean(),6)} ({round(cv_results.std(), 6)})")
